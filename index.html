class CodersMeet {
  constructor() {
    this.community = [];
    this.projects = [];
    this.events = [];
  }

  connect(developer) {
    this.community.push(developer);
    return `Welcome ${developer.name}!`;
  }

  createProject(name, tech) {
    const project = new Project(name, tech);
    this.projects.push(project);
    return project;
  }

  hostEvent(title, date) {
    const event = new Event(title, date);
    this.events.push(event);
    return event;
  }

  get mission() {
    return "Empowering coders to 
    build the future together";
  }
}
